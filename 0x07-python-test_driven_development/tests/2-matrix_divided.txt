"""
Test Documentation for matrix_divided Function
"""

def test_matrix_divided():
    """
    Test cases for the matrix_divided function.
    """
    # Test case 1: Basic division with integers
    matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
    result = matrix_divided(matrix, 2)
    expected_result = [[0.5, 1.0, 1.5], [2.0, 2.5, 3.0], [3.5, 4.0, 4.5]]
    assert result == expected_result, f"Expected {expected_result}, but got {result}"

    # Test case 2: Basic division with floating-point numbers
    matrix = [[1.5, 2.5, 3.5], [4.5, 5.5, 6.5], [7.5, 8.5, 9.5]]
    result = matrix_divided(matrix, 2)
    expected_result = [[0.75, 1.25, 1.75], [2.25, 2.75, 3.25], [3.75, 4.25, 4.75]]
    assert result == expected_result, f"Expected {expected_result}, but got {result}"

    # Test case 3: Division by zero
    try:
        matrix_divided(matrix, 0)
    except ZeroDivisionError as e:
