"""
Test Documentation for say_my_name Function
"""

def test_say_my_name():
    """
    Test cases for the say_my_name function.
    """
    # Test case 1: Basic usage with both first and last names
    first_name = "John"
    last_name = "Doe"
    result = say_my_name(first_name, last_name)
    expected_result = f"My name is {first_name} {last_name}"
    assert result == expected_result, f"Expected {expected_result}, but got {result}"

    # Test case 2: Basic usage with only the first name
    first_name = "Jane"
    result = say_my_name(first_name)
    expected_result = f"My name is {first_name} "
    assert result == expected_result, f"Expected {expected_result}, but got {result}"

    # Test case 3: Non-string first name
    try:
        say_my_name(123, last_name)
    except TypeError as e:
        assert str(e) == "first_name must be a string", f"Expected 'first_name must be a string', but got {e}"

    # Test case 4: Non-string last name
    try:
        say_my_name(first_name, 456)
    except TypeError as e:
        assert str(e) == "last_name must be a string", f"Expected 'last_name must be a string', but got {e}"

    print("All test cases passed!")

# Run the test cases
test_say_my_name()
